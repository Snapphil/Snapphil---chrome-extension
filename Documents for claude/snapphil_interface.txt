    <script>
        // Tab switching functionality
        document.addEventListener('DOMContentLoaded', function() {
            const tabItems = document.querySelectorAll('.nav-item');
            const tabContents = document.querySelectorAll('.tab-content');

            tabItems.forEach(item => {
                item.addEventListener('click', function() {
                    // Remove active class from all tabs
                    tabItems.forEach(tab => tab.classList.remove('active'));
                    tabContents.forEach(content => content.classList.remove('active'));

                    // Add active class to clicked tab
                    this.classList.add('active');
                    const tabId = this.getAttribute('data-tab');
                    document.getElementById(tabId).classList.add('active');
                });
            });

            // Resume upload functionality
            const resumeUpload = document.getElementById('resumeUpload');
            const fileInput = document.getElementById('resume-file-input');
            const parsingIndicator = document.querySelector('.parsing-indicator');
            const uploadStatus = document.getElementById('upload-status');
            const fileName = document.getElementById('file-name');
            const extractionStatus = document.getElementById('extraction-status');
            
            // Show file picker when clicking the upload area
            resumeUpload.addEventListener('click', function() {
                fileInput.click();
            });
            
            // Handle file selection
            fileInput.addEventListener('change', function(e) {
                if (this.files && this.files[0]) {
                    const file = this.files[0];
                    
                    // Show the file name and the parsing indicator
                    fileName.textContent = file.name;
                    parsingIndicator.classList.add('visible');
                    
                    // Reset any previous animations
                    uploadStatus.classList.remove('parsed');
                    document.getElementById('success-pulse').classList.remove('animate');
                    document.getElementById('success-check').classList.remove('visible');
                    document.getElementById('parsed-status').style.display = 'none';
                    document.getElementById('extraction-status').style.display = 'block';
                    
                    // Show upload status
                    setTimeout(() => {
                        uploadStatus.classList.add('visible');
                        extractionStatus.textContent = '• Extracting work experience';
                        
                        // Start parsing simulation
                        setTimeout(() => {
                            extractionStatus.textContent = '• Identifying skills and qualifications';
                            
                            setTimeout(() => {
                                extractionStatus.textContent = '• Analyzing education history';
                                
                                // Final success state
                                setTimeout(() => {
                                    // Hide the extraction status and show success message
                                    document.getElementById('extraction-status').style.display = 'none';
                                    document.getElementById('parsed-status').style.display = 'block';
                                    
                                    // Add the parsed animation class
                                    uploadStatus.classList.add('parsed');
                                    
                                    // Animate the success pulse
                                    document.getElementById('success-pulse').classList.add('animate');
                                    
                                    // Show the checkmark
                                    setTimeout(() => {
                                        document.getElementById('success-check').classList.add('visible');
                                    }, 700);
                                    
                                    // Hide the parsing indicator
                                    parsingIndicator.classList.remove('visible');
                                }, 500);
                            }, 500);
                        }, 500);
                    }, 300);
                }
            });
            
            // Drag and drop functionality
            resumeUpload.addEventListener('dragover', function(e) {
                e.preventDefault();
                this.style.background = 'rgba(50, 50, 50, 0.5)';
                this.style.borderColor = 'rgba(255, 255, 255, 0.3)';
            });
            
            resumeUpload.addEventListener('dragleave', function(e) {
                e.preventDefault();
                this.style.background = 'rgba(30, 30, 30, 0.4)';
                this.style.borderColor = 'rgba(255, 255, 255, 0.1)';
            });
            
            resumeUpload.addEventListener('drop', function(e) {
                e.preventDefault();
                this.style.background = 'rgba(30, 30, 30, 0.4)';
                this.style.borderColor = 'rgba(255, 255, 255, 0.1)';
                
                if (e.dataTransfer.files && e.dataTransfer.files[0]) {
                    fileInput.files = e.dataTransfer.files;
                    const event = new Event('change', { bubbles: true });
                    fileInput.dispatchEvent(event);
                }
            });
            
            // Personal info toggle
            const personalInfoToggle = document.querySelector('.personal-info-toggle');
            const personalInfoFields = document.querySelector('.personal-info-fields');
            const chevron = document.querySelector('.chevron');
            
            personalInfoToggle.addEventListener('click', function() {
                personalInfoFields.classList.toggle('expanded');
                chevron.classList.toggle('down');
            });
            
            // Context area interactivity
            const contextInput = document.querySelector('.context-input');
            const charCount = document.getElementById('char-count');
            const contextTags = document.querySelectorAll('.context-tag');
            
            // Character counter
            contextInput.addEventListener('input', function() {
                const length = this.value.length;
                charCount.textContent = length;
                
                // Change color if approaching limit
                if (length > 400) {
                    charCount.parentElement.style.color = 'rgba(255, 180, 0, 0.8)';
                } else if (length > 480) {
                    charCount.parentElement.style.color = 'rgba(255, 100, 100, 0.8)';
                } else {
                    charCount.parentElement.style.color = 'rgba(255, 255, 255, 0.5)';
                }
                
                // Auto resize textarea
                this.style.height = 'auto';
                this.style.height = (this.scrollHeight) + 'px';
            });
            
            // Quick tag insertion
            contextTags.forEach(tag => {
                tag.addEventListener('click', function() {
                    const tagText = this.getAttribute('data-tag');
                    contextInput.value += (contextInput.value ? '\n' : '') + tagText + ': ';
                    contextInput.focus();
                    
                    // Trigger input event to update character count
                    const event = new Event('input', {
                        bubbles: true,
                        cancelable: true,
                    });
                    contextInput.dispatchEvent(event);
                });
            });
            
            // Smooth focus animation
            contextInput.addEventListener('focus', function() {
                this.parentElement.style.transform = 'translateY(-2px)';
            });
            
            contextInput.addEventListener('blur', function() {
                this.parentElement.style.transform = 'translateY(0)';
            });
            
            // FAQ accordion functionality
            const faqItems = document.querySelectorAll('.faq-item');
            
            faqItems.forEach(item => {
                const question = item.querySelector('.faq-question');
                
                question.addEventListener('click', () => {
                    // Toggle active class on the clicked item
                    const isActive = item.classList.contains('active');
                    
                    // Close all FAQ items first
                    faqItems.forEach(faq => {
                        faq.classList.remove('active');
                    });
                    
                    // If the clicked item wasn't active, open it
                    if (!isActive) {
                        item.classList.add('active');
                    }
                });
            });
        });
    </script>
